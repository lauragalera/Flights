113 void Airlines::manageIATARepeated(Airline * newAirline) {
114 
115  string IATA = newAirline->getIATA();
116  map<string, Airline *>::iterator it = _mapIATAAirlines.find(IATA);
117 
118  if (it != _mapIATAAirlines.end()) { //codi IATA ja existent
119 
120  bool activeOldOne = (it->second)->isActive();
121 
122  if (newAirline->isActive() and !activeOldOne) { //nova activa, primera inactiva
123  _mapIATAAirlines[IATA] = newAirline; //substituim
124  if (_IATARepeated.find(IATA) != _IATARepeated.end()) _IATARepeated.erase(IATA); //IATA ja no compta com a repetit
125  } else if ((newAirline->isActive() and activeOldOne) or (!(newAirline->isActive()) and !activeOldOne)) { //nova i primera actives o nova i primera inactives
126  _IATARepeated.insert(IATA); //afegim el IATA a repetits
127  }
128  } else _mapIATAAirlines[IATA] = newAirline; //codi IATA no està al map, afegim.
