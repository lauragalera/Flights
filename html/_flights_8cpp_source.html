122 list<pair<Date, int>> Flights::daysMoreDelay(const vector<string> &dataRead, const Airports & airports) const {
123 
124  map<Date, int> countDate;
125  if (dataRead.size() == 1) { //per a tots els aeroports
126  for (pair<Date, list < Flight*>> i : _mapDate) {
127  countDate[i.first] = 0;
128  for (Flight * f : i.second) { //recorrem els vols d'una data
129  if (f->getDelay('D') > 0)
130  countDate[f->getDate()] += f->getDelay('D'); //sumem retard sortida

132  countDate[f->getDate()] += f->getDelay('A'); //sumem retard arribada
133  }
134  }
135  } else { //només per a un subconjunt d'aeroports
136  for (unsigned i = 1; i < dataRead.size(); i++) { //recorrem cada aeroport
137  if (!airports.airportExists(-1, dataRead[i]))
138  cerr << "ERROR: " + dataRead[i] + " is an unknown IATA airport code \n\n"; //missatge d'error, no existeix.
139  else {
140  map<string, list<Flight * > >::const_iterator it = _mapIATAAirport.find(dataRead[i]);
141  if (it != _mapIATAAirport.end()) { //si l'aeroport ha tingut vols
142  for (Flight * f : it->second) { //recorrem tots els vols de l'aeroport.
143  int delay = 0;
144  if (f->getIATAAirport('D') == dataRead[i] and f->getDelay('D') > 0) { //si és l'aeroport de sortida i el retard es positiu
145  delay = f->getDelay('D');
146  } else if (f->getIATAAirport('A') == dataRead[i] and f->getDelay('A') > 0) { //si és l'aeroport d'arribada i el retard es positiu
147  delay = f->getDelay('A');

149  if (delay > 0) { //en aquella data hi ha hagut retard

151  countDate[f->getDate()] = 0;

153  }
154  }
155  }
156  }
157  }
158  }

160  set<pair<int, Date>> orderedDelayDay;
161  for (pair<Date, int> j : countDate) //set per ordenar segons retard

170  dayMoreDelay.push_back(make_pair(itR->second, itR->first));

172  }
